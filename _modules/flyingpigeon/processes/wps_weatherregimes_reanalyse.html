def ocgis_call_wrap(tmargs):
    _z=tmargs[0]
    _origvar=tmargs[1]
    _level=tmargs[2]
    _bbox=tmargs[3]
    _plev=[int(_level), int(_level)]
    _pref='levdom_'+basename(_z)[0:-3]

    _tmpf=call(resource=_z, variable=_origvar, level_range=_plev, geom=_bbox,
                spatial_wrapping='wrap', prefix=_pref)

    return _tmpf
